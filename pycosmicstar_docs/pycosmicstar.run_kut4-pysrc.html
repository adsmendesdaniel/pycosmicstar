50          K0 = h * F(x, y) 
51          K1 = h * F(x + h / 2.0, y + K0 / 2.0) 
52          K2 = h * F(x + h / 2.0, y + K1 / 2.0) 
53          K3 = h * F(x + h, y + K2) 
54          #print 'run',K0,K1,K2,K3 
55          return (K0 + 2.0 * K1 + 2.0 * K2 + K3) / 6.0
